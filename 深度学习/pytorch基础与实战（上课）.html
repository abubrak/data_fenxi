# 初始化参数
k=torch.tensor([0.0],requires_grad=True)
b=torch.tensor([0.0],requires_grad=True)
y_pred=k*x+b
def loss(y,y_pred):
    return torch.sum(torch.square(y-y_pred))/len(y)
err=loss(y,y_pred)
# 反向梯度回传
err.backward()
# 设置学习速率
lr=0.02
# 更新参数
k.data-=lr*k.grad        # k=k-lr*k.grad
b.data-=lr*b.grad
print(k.grad)
print(b.grad)
# 梯度清零
k.grad.data.zero_()
b.grad.data.zero_()
print(k.grad)
print(b.grad)

    k.data-=lr*k.grad       
    b.data-=lr*b.grad
    k.grad.data.zero_()
    b.grad.data.zero_()
    Err.append(err.item())

BPmodel=nn.Sequential(
    nn.Linear(3,100),      # 输入神经元个数，隐层神经元个数
    nn.Softmax(),          # 激活函数
    nn.Linear(100,1))      # 隐层神经元个数，输出神经元个数
loss_mse=nn.MSELoss()      # 损失函数
optimizer=optim.SGD(BPmodel.parameters(),lr=0.001)    # 优化函数（随机梯度下降）
