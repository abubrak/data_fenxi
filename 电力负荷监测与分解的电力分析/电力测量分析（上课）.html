def fun(x):
    x=x.values
    x=x.reshape(128,1)
    kmodel=KMeans(n_clusters=2)
    kmodel.fit(x)
    cluster_data=kmodel.cluster_centers_
    low=cluster_data.min()    # 波谷
    up=cluster_data.max()    # 波峰
    # 获取波峰波谷对应的样本数
    low_num=(kmodel.labels_==cluster_data.argmin()).sum()
    up_num=(kmodel.labels_==cluster_data.argmax()).sum()
    return low,up,low_num,up_num
